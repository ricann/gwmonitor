drop procedure if exists del_data;

delimiter //
create procedure del_data()
begin
	delete from scalardata where time<date_sub(curdate(),interval 8 days);
end //

drop event if exists del_event//

create event del_event
on schedule
every 1 day
starts '2014-04-01 18:00:00'
on completion preserve enable
do call del_data() //

delimiter ;

show events;
show variables like '%sche%';
